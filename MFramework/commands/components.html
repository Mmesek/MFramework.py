 38async def interaction_create(client: 'Bot', interaction: Interaction):
 39    '''Called after receiving event INTERACTION_CREATE from Discord
 40    Reacts only to Components (Buttons)'''
 41    from MFramework import Context
 42    if interaction.type != Interaction_Type.MESSAGE_COMPONENT and interaction.type != Interaction_Type.MODAL_SUBMIT:
 43        return
 44    name, data = interaction.data.custom_id.split("-", 1)
 45    f = components.get(name, None)
 46    ctx = Context(client.cache, client, interaction)
 47    if not f:
 48        log.debug("Component %s not found", name)
 49        return
 50
 51    if f.auto_deferred:
 52        await interaction.deferred(f.private_response)
 53    
 54    if interaction.data.component_type == Component_Types.SELECT_MENU:
 55        not_selected = []
 56        for row in interaction.message.components:
 57            for select_component in filter(lambda x: x.type == Component_Types.SELECT_MENU, row.components):
 58                if any(v in [_v.value for _v in select_component.options] for v in interaction.data.values):
 59                    for value in select_component.options:
 60                        if value.value not in interaction.data.values:
 61                            not_selected.append(value)
 62        return await run_function(f, ctx, data=data, values=interaction.data.values, not_selected=not_selected)
 63    elif interaction.type == Interaction_Type.MODAL_SUBMIT:
 64        inputs = {}
 65        for row in interaction.data.components:
 66            for text_input in filter(lambda x: x.type == Component_Types.TEXT_INPUT, row.components):
 67                inputs[text_input.custom_id.split("-", 1)[-1]] = text_input.value
 68        return await run_function(f, ctx, data=data, inputs=inputs)
 69
 70    return await run_function(f, ctx, data=data)
