 96class DatabaseCooldown(Cooldown):

 98    def __init__(self, ctx: Context, cooldown: timedelta, cooldown_type: str, table: Any, **kwargs) -> None:
 99        self.table = table
100        super().__init__(ctx=ctx, cooldown=cooldown, cooldown_type=cooldown_type, **kwargs)

102    def last_action(self) -> datetime:
103        session = self.ctx.db.sql.session()
104        r = session.query(self.table).filter(self.table.user_id == self.ctx.user_id, self.table.server_id == self.ctx.guild_id).first()
105        return r.timestamp if r else None
106    
107    def add_cooldown(self) -> None:
108        session = self.ctx.db.sql.session()
109        r = session.query(self.table).filter(self.table.user_id == self.ctx.user_id, self.table.server_id == self.ctx.guild_id).first()
110        if r:
111            r.timestamp = datetime.now(tz=timezone.utc)
112        else:
113            session.add(self.table(user_id=self.ctx.user_id, server_id=self.ctx.guild_id))

116class Tokens(Cooldown):
117    '''Simple Tokens counter before applying cooldown'''
118    def __init__(self, ctx: Context, capacity: int, cooldown: timedelta, cooldown_type: str, **kwargs) -> None:
119        self.capacity: int = capacity
120        self._count = 0
121        super().__init__(ctx, cooldown, cooldown_type, **kwargs)
122        v = self._get_cooldown()
123        self._last_action = v[0]
124        self._count = v[1]
125    def _get_cooldown(self) -> Any:
126        return super()._get_cooldown() or (None, 0)
127
128    @property
129    def last_action(self) -> datetime:
130        return self._last_action
131    
132    def add_cooldown(self, value=None) -> None:
133        return super().add_cooldown(value=(self._count, self.now()))
134
135    @property
136    def on_cooldown(self) -> bool:
137        if not super().on_cooldown:
138            # Restart window
139            self._count = self.capacity
140
141        if not self._count:
142            # No more free tokens, cooldown
143            return True
144
145        # No cooldown, decrease remaining tokens
146        self._count -= 1
147        return False

149class SlidingWindow(Tokens):
150    '''Remaining Tokens affected by usage during cooldown's previous window'''
151    @property
152    def on_cooldown(self) -> bool:
153        if not super(Tokens).on_cooldown:
154            # Restart window
155            self._previous = self._count
156            self._count = 0
157
158        t = self.remaining / self.cooldown
159        est = self._previous * t + self._count
160
161        if est > self.capacity:
162            # Usage within this and previous window suggest *probably* depleted tokens and therefore Cooldown
163            return True
164
165        # No cooldown, usage counter increased
166        self._count += 1
167        return False

170def cooldown(
171        rate: int=1, 
172        bucket: Bucket = Bucket.USER, 
173        scope: str = None, 
174        *, 
175        seconds: int=0, minutes: int=0, hours: int=0, days: int=0, weeks: int=0, 
176        logic: Cooldown=Cooldown, 
177        delta: timedelta = None, 
178        **cooldown_kwargs
179    ):
180    '''Applies a cooldown on command.
181
182    Parameters
183    ----------
184    rate:
185        After how many uses cooldown should be applied
186    bucket:
187        Group which cooldown is applied to (User, Guild, Channel, Role, Group). Default: User
188    scope:
189        Group of cooldowns (Global, Functions, Function). Default: Function
190    logic:
191        Backend class which handles Cooldown calculation on command invocation
192    delta:
193        Delta to be used as a cooldown value
194    cooldown_kwargs:
195        Keyword arguments that should be passed to Cooldown (logic) class
196    Use it with callable function accepting interaction returning boolean for conditional execution and datetime object with last execution timestamp for cooldown calculation'''
197    def inner(f):
198        @wraps(f)
199        def wrapped(ctx: 'Context'= None, **kwargs):
200            _cooldown = delta or timedelta(days=days, seconds=seconds, minutes=minutes, hours=hours, weeks=weeks)
201            c = logic(ctx=ctx, cooldown=_cooldown, cooldown_type=scope or f.__name__, func_args = kwargs, bucket=bucket, capacity=rate, **cooldown_kwargs)
202            if not c.on_cooldown:
203                c.add_cooldown()
204                return f(ctx=ctx, **kwargs)
205            from .exceptions import CooldownError
206            raise CooldownError(c.remaining)
207        return wrapped
208    return inner
