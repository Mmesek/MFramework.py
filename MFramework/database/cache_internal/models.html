 11class Collection(MutableMapping[KT, VT]):
 12    """Base Collection.
 13
 14    Default ID is composed from Class name and `.id` attribute of cached object"""
 15
 16    _cache: Redis
 17    _expire: timedelta = None
 18    _cls: str = VT
 19
 20    def __init__(self, cache: Union[Dictionary, Redis] = None, value_factory: VT = None) -> None:
 21        self._cache = cache or Dictionary()
 22        if value_factory:
 23            self._cls = value_factory
 24
 25    # Dictionary-like usage
 26    async def __setitem__(self, key: KT, value: VT) -> None:
 27        await self._cache.add(self._combine(key), value, self._expire)
 28
 29    async def __len__(self) -> int:
 30        return await self._cache.db_size()
 31
 32    async def __delitem__(self, key: KT) -> None:
 33        await self._cache.delete(self._create_id(key))
 34
 35    def items(self) -> tuple[tuple[KT, VT]]:
 36        if type(self._cache) is Dictionary:
 37            return self._cache.items()
 38        return {}
 39
 40    def keys(self) -> list[KT]:
 41        return self._cache.keys()
 42
 43    def values(self):
 44        if type(self._cache) is Dictionary:
 45            return self._cache.values()
 46        return iter({})
 47
 48    def __iter__(self):
 49        if type(self._cache) is Dictionary:
 50            return iter(self._cache)
 51        return iter({})
 52
 53    async def update(self, obj: VT | list[VT]) -> str:
 54        """Update current object(s) in cache"""
 55        if type(obj) is list:
 56            for i in obj:
 57                await self.update(i)
 58            return
 59        return await self._cache.update(self._create_id(obj), obj)
 60
 61    async def pop(self, *args, default: Any=None) -> int | Any:
 62        return await self._cache.delete(*args) or default
 63
 64    async def __contains__(self, o: VT) -> bool:
 65        return await self._cache.has(o)
 66
 67    async def __getitem__(self, id: Snowflake) -> VT:
 68        r = await self._cache.get(self._combine(id))
 69        if type(r) is dict:
 70            return self._cls(**r)
 71        return r
 72
 73    async def get(self, id: Snowflake, default=None) -> VT | Any:
 74        return await self._cache.get(self._combine(id)) or default
 75
 76    # Internal
 77    def _create_id(self, obj: VT) -> str:
 78        """Creates ID by combining object ID with class's name"""
 79        if type(self._cache) is Dictionary:
 80            return obj.id
 81        return self._combine(obj.id)
 82
 83    def _combine(self, *args) -> str:
 84        """Combines args with class's name into a dotted key"""
 85        if type(self._cache) is Dictionary:
 86            return args[-1]
 87        return ".".join([self.__class__.__name__] + [str(i) for i in args])
 88
 89    # External API
 90    async def from_list(self, iterable: list[VT], guild_id: Snowflake=None) -> None:
 91        """Fill cache from list"""
 92        for i in iterable:
 93            if guild_id:
 94                i.guild_id = guild_id
 95            await self.store(i)
 96        return self

 99        """Store a single item"""
100        await self._cache.add(self._create_id(obj), obj, expire_time=self._expire)
101
102    async def has(self, name: str) -> bool:
103        """Check if provided key exisits in cache"""
104        return True if await self._cache.exists(self._combine(name)) else False

106    async def delete(self, obj_id: VT) -> int:
107        """Delete from Cache"""
