 47    username = "Message Log"
 48    def set_metadata(self, msg: MFramework.Message) -> MFramework.Embed:
 49        embed = MFramework.Embed()
 50        embed.setDescription(msg.content)
 51        embed.setTimestamp(msg.timestamp)
 52        embed.setFooter(f"ID: {msg.author.id}")
 53        embed.setAuthor(f"{msg.author.username}#{msg.author.discriminator}", None, msg.author.get_avatar())
 54        return embed
 55    def user_in_footer(self, embed: MFramework.Embed, msg: MFramework.Message) -> MFramework.Embed:
 56        return embed.setFooter(text=msg.author.username, icon_url=msg.author.get_avatar())

 58    def get_cached_message(self, key: MFramework.Snowflake) -> MFramework.Message:
 59        return self.bot.cache[self.guild_id].messages[key]
 60        #return self.bot.cache[self.guild_id].messages.get(key, None) #getMessage(message_id, channel_id)
 61    def cached_message(self, msg: MFramework.Message) -> MFramework.Embed:
 62        cached = self.get_cached_message(f"{msg.guild_id}.{msg.channel_id}.{msg.id}")
 63        if cached:
 64            embed = self.set_metadata(cached)
 65            if cached.attachments != None:
 66                attachments = ''
 67                for attachment in cached.attachments:
 68                    attachments += attachment.filename + "\n"
 69                if attachments != '':
 70                    embed.addField("Attachments:", attachments)
 71            return embed

102class Voice(Log):
103    username = "Voice Log"
104    async def log(self, data: MFramework.Voice_State, channel: MFramework.Snowflake='', after: int=None) -> MFramework.Message:
105        string = f'<@{data.user_id}> '
106        if channel != '' and data.channel_id != channel and data.channel_id != 0:
107            string += f'moved from <#{channel}> to '
108            status = '|'
109            channel = data.channel_id
110        elif data.channel_id == 0:
111            string += 'left '
112            status = '-'
113        else:
114            string += 'joined '
115            status = '+'
116            channel = data.channel_id
117        if channel == -1:
118            channel = self.bot.cache[data.guild_id].afk_channel
119        string += f'<#{channel}>'
120        if after is not None and after > 0:
121            from mlib.localization import secondsToText
122            string += f" after {secondsToText(after)}"
123        await self._log(string, username=self.username+ f' [{status}]')

125class Guild_Member_Update(Log):
126    username = "Member Update Log"
127    async def log(self, data: MFramework.Guild_Member_Update) -> MFramework.Message:
128        ctx = self.bot
129        if data.user.id not in ctx.cache[data.guild_id].members:
130            return
131        c = ctx.cache[data.guild_id].members[data.user.id]
132        diff = set(c.roles) ^ set(data.roles)
133        
134        if len(diff) == 0:
135            return # Something else
136        elif any(i in data.roles for i in diff):
137            case = 'added'
138        else:
139            case = 'removed'
140        roles = ""
141        for i in diff:
142            if i == ctx.cache[data.guild_id].voice_link:
143                if len(diff) == 1:
144                    return
145                continue
146            roles += f"<@&{i}> "
147        s =''
148        if len(diff) > 1:
149            s = 's'
150        string = f"<@{data.user.id}> role{s} {case}: {roles}"
151        await self._log(string)
152        ctx.cache[data.guild_id].members[data.user.id] = data

154class Nitro_Change(Guild_Member_Update):
155    username = "Nitro Log"
156    async def log(self, data: MFramework.Guild_Member_Update) -> MFramework.Message:
157        ctx = self.bot
158        if data.user.id in ctx.cache[data.guild_id].members:
159            c = ctx.cache[data.guild_id].members[data.user.id]
160            diff = set(c.roles) ^ set(data.roles)
161            if len(diff) == 0:
162                return
163            elif any(i in data.roles for i in diff) and data.premium_since is not None:
164                case = 'started boosting'
165                s = True
166            else:
167                s = False
168                case = 'stopped boosting'
169            booster = False
170            for i in diff:
171                if i in ctx.cache[data.guild_id].groups[MFramework.Groups.NITRO]:
172                    booster = True
173                    break
174            if booster:
175                await self._log(f"<@{data.user.id}> {case}")
176                ctx.cache[data.guild_id].members[data.user.id] = data
177                return s
